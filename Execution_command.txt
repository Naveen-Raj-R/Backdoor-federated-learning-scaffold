For Trigger-based Backdoor Attacks:

# Neural Cleanse defense with IID data
python main.py --dist_type iid --attack_type trigger --defense_type neural_cleanse

# Fine Pruning defense with IID data
python main.py --dist_type iid --attack_type trigger --defense_type fine_pruning

# Mode Connectivity (MCR) defense with IID data
python main.py --dist_type iid --attack_type trigger --defense_type mcr

# Same attacks with Non-IID data (with different Dirichlet alpha values)
python main.py --dist_type non-iid --attack_type trigger --defense_type neural_cleanse --dirichlet_alpha 0.1
python main.py --dist_type non-iid --attack_type trigger --defense_type fine_pruning --dirichlet_alpha 0.5
python main.py --dist_type non-iid --attack_type trigger --defense_type mcr --dirichlet_alpha 0.9

Additional Parameters that can be adjusted

# Change number of clients
python main.py --num_clients 10 --attack_type trigger --defense_type neural_cleanse

# Modify training parameters
python main.py --num_rounds 100 --local_epochs 20 --batch_size 64 --learning_rate 0.001 --defense_type neural_cleanse

# Change output directory
python main.py --output_dir ./custom_output --defense_type neural_cleanse

# Combine multiple parameters
python main.py --dist_type non-iid --attack_type semantic --defense_type mcr --num_clients 8 --num_rounds 75 --dirichlet_alpha 0.4 --batch_size 128 --learning_rate 0.005

Base line runs without Defense
# IID data with trigger-based attack
python main.py --dist_type iid --attack_type trigger --defense_type none

# Non-IID data with semantic attack
python main.py --dist_type non-iid --attack_type semantic --defense_type none --dirichlet_alpha 0.5
